STEP 2: Navigate to project directory (try these)

===========================================
STEP 3: (Optional) Backup existing database
===========================================
# Try multiple possible database locations
cp ../../database.sqlite ../../database.sqlite.backup.$(date +%Y%m%d_%H%M%S) 2>/dev/null || \
cp database.sqlite database.sqlite.backup.$(date +%Y%m%d_%H%M%S) 2>/dev/null || \
echo "No database to backup or already backed up"

===========================================
STEP 4: Delete existing database
===========================================
rm -f ../../database.sqlite
rm -f database.sqlite
rm -f ./database.sqlite
rm -f ~/project/database.sqlite

===========================================
STEP 5: Run the new seed script
===========================================
# First verify the file exists
ls -la config/seed-new.js

# If file exists, run it directly (simplest method):
node config/seed-new.js

# If that doesn't work, try from the config directory:
cd config
node seed-new.js
cd ..

# OR if you need to use require, make sure you're in the right directory:
cd ~/project/src
node -e "const path = require('path'); const seedPath = path.join(__dirname, 'config', 'seed-new'); require(seedPath).seedDatabase();"
